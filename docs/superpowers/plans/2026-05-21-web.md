# Web Frontend Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship `@buddy/web` — a Vite + React 18 + TypeScript SPA that lets a user pick a topic, open/create/rename/delete/export conversations, send queries, watch streamed answers with citations + reasoning, and preview cited PDF pages. By end of this plan a user can build a topic via `apps/build-index`, start `apps/serve`, open the browser, and chat end-to-end.

**Architecture:**
- React 18 (function components + hooks). No class components. No global store.
- `@tanstack/react-query` for server state (topics, docs, conversations, messages). Local `useState/useReducer` for streaming buffer.
- Tailwind CSS + `shadcn/ui` primitives (button, dialog, dropdown-menu, scroll-area, separator) installed locally via `shadcn add` so source lives in repo.
- React Router v6 for two routes: `/` (TopicSelect) and `/t/:topic` (Chat).
- SSE consumed via `fetch().body.getReader()` + manual parser (text/event-stream).
- Vitest + React Testing Library + jsdom for smoke tests.
- Vite dev server proxies `/api/*` → `http://localhost:${PORT}`. Prod build → `dist/` served by Hono via `staticRoutes`.

**Tech Stack:** vite, @vitejs/plugin-react, react, react-dom, react-router-dom, @tanstack/react-query, tailwindcss, postcss, autoprefixer, lucide-react, clsx, vitest, @testing-library/react, @testing-library/user-event, jsdom, msw (mock SSE for tests).

```ts
const DAY = 24 * 60 * 60 * 1000;

export function truncate(s: string, n: number): string {
  return s.length <= n ? s : s.slice(0, Math.max(0, n - 1)) + '…';
}

export function formatRange(range: [number, number]): string {
  return range[0] === range[1] ? `p.${range[0]}` : `p.${range[0]}-${range[1]}`;
}

export interface DateGroup<T> { label: 'Today' | 'Yesterday' | 'This week' | 'Older'; items: T[]; }

export function groupByDate<T>(items: T[], getDate: (item: T) => number): DateGroup<T>[] {
  const now = new Date();
  const startOfToday = new Date(now.getFullYear(), now.getMonth(), now.getDate()).getTime();
  const startOfYesterday = startOfToday - DAY;
  const startOfWeek = startOfToday - 6 * DAY;

  const buckets: Record<DateGroup<T>['label'], T[]> = {
    Today: [], Yesterday: [], 'This week': [], Older: [],
  };
  for (const it of items) {
    const t = getDate(it);
    if (t >= startOfToday) buckets.Today.push(it);
    else if (t >= startOfYesterday) buckets.Yesterday.push(it);
    else if (t >= startOfWeek) buckets['This week'].push(it);
    else buckets.Older.push(it);
  }
  return (['Today', 'Yesterday', 'This week', 'Older'] as const)
    .filter((k) => buckets[k].length > 0)
    .map((k) => ({ label: k, items: buckets[k] }));
}
```

```ts
import {
  topicSummarySchema, docSummarySchema, conversationSummarySchema, messageSchema,
  type TopicSummary, type DocSummary, type ConversationSummary, type Message,
} from '@buddy/shared';
import { z } from 'zod';

export class ApiError extends Error {
  status: number;
  constructor(status: number, message: string) { super(message); this.status = status; }
}

async function req<T>(url: string, init: RequestInit | undefined, schema: z.ZodType<T>): Promise<T> {
  const res = await fetch(url, { ...init, headers: { 'content-type': 'application/json', ...(init?.headers ?? {}) } });
  if (!res.ok) {
    const text = await res.text();
    throw new ApiError(res.status, text || res.statusText);
  }
  const body = (await res.json()) as unknown;
  return schema.parse(body);
}

export const api = {
  topics: (): Promise<TopicSummary[]> =>
    req('/api/topics', undefined, z.array(topicSummarySchema)),

  docs: (topic: string): Promise<DocSummary[]> =>
    req(`/api/topics/${encodeURIComponent(topic)}/docs`, undefined, z.array(docSummarySchema)),

  conversations: (topic: string): Promise<ConversationSummary[]> =>
    req(`/api/conversations?topic=${encodeURIComponent(topic)}`, undefined, z.array(conversationSummarySchema)),

  createConversation: (topic: string, title?: string): Promise<{ id: string }> =>
    req('/api/conversations', {
      method: 'POST',
      body: JSON.stringify({ topic, ...(title ? { title } : {}) }),
    }, z.object({ id: z.string() })),

  renameConversation: (id: string, title: string): Promise<{ ok: boolean }> =>
    req(`/api/conversations/${encodeURIComponent(id)}`, {
      method: 'PATCH', body: JSON.stringify({ title }),
    }, z.object({ ok: z.boolean() })),

  deleteConversation: (id: string): Promise<{ ok: boolean }> =>
    req(`/api/conversations/${encodeURIComponent(id)}`, { method: 'DELETE' }, z.object({ ok: z.boolean() })),

  messages: (cid: string): Promise<Message[]> =>
    req(`/api/conversations/${encodeURIComponent(cid)}/messages`, undefined, z.array(messageSchema)),

  pdfPageUrl: (topic: string, docId: string, page: number, scale = 2): string =>
    `/api/pdf/${encodeURIComponent(topic)}/${encodeURIComponent(docId)}?page=${page}&scale=${scale}`,
};
```

```ts
export interface SseEvent { event: string; data: unknown; }

export async function* parseSseStream(
  reader: ReadableStreamDefaultReader<Uint8Array>,
): AsyncIterable<SseEvent> {
  const decoder = new TextDecoder('utf-8');
  let buf = '';
  while (true) {
    const { value, done } = await reader.read();
    if (done) break;
    buf += decoder.decode(value, { stream: true });

    let sep: number;
    while ((sep = buf.indexOf('\n\n')) !== -1) {
      const raw = buf.slice(0, sep);
      buf = buf.slice(sep + 2);
      if (!raw.trim()) continue;
      let event = 'message';
      const dataLines: string[] = [];
      for (const line of raw.split('\n')) {
        if (line.startsWith(':') || line.length === 0) continue;
        if (line.startsWith('event:')) event = line.slice(6).trim();
        else if (line.startsWith('data:')) dataLines.push(line.slice(5).trim());
      }
      if (dataLines.length === 0) continue;
      const dataStr = dataLines.join('\n');
      let parsed: unknown;
      try { parsed = JSON.parse(dataStr); } catch { parsed = dataStr; }
      yield { event, data: parsed };
    }
  }
}

export interface OpenChatStreamOpts {
  conversationId: string;
  query: string;
  signal?: AbortSignal;
}

export async function openChatStream(opts: OpenChatStreamOpts): Promise<AsyncIterable<SseEvent>> {
  const res = await fetch('/api/chat/stream', {
    method: 'POST',
    headers: { 'content-type': 'application/json' },
    body: JSON.stringify({ conversation_id: opts.conversationId, query: opts.query }),
    signal: opts.signal,
  });
  if (!res.ok || !res.body) {
    throw new Error('chat stream failed: ' + res.status);
  }
  return parseSseStream(res.body.getReader());
}
```

```ts
import { useCallback, useReducer, useRef } from 'react';
import { useQueryClient } from '@tanstack/react-query';
import type { Citation, ReasoningTrace } from '@buddy/shared';
import { openChatStream } from '../api/sse.js';
import { messagesKey } from './messages.js';
import { conversationsKey } from './conversations.js';

export interface PendingMessage {
  content: string;
  citations: Citation[];
  trace: ReasoningTrace | null;
  done: boolean;
  error: string | null;
}

type Action =
  | { type: 'start' }
  | { type: 'token'; delta: string }
  | { type: 'citations'; citations: Citation[] }
  | { type: 'trace'; trace: ReasoningTrace }
  | { type: 'done' }
  | { type: 'error'; message: string }
  | { type: 'reset' };

const initial: PendingMessage = { content: '', citations: [], trace: null, done: false, error: null };

function reduce(state: PendingMessage | null, a: Action): PendingMessage | null {
  switch (a.type) {
    case 'start': return { ...initial };
    case 'token': return state ? { ...state, content: state.content + a.delta } : state;
    case 'citations': return state ? { ...state, citations: a.citations } : state;
    case 'trace': return state ? { ...state, trace: a.trace } : state;
    case 'done': return state ? { ...state, done: true } : state;
    case 'error': return state ? { ...state, error: a.message, done: true } : state;
    case 'reset': return null;
  }
}

export interface UseChatStreamOpts { conversationId: string; topic: string; }

export function useChatStream(opts: UseChatStreamOpts) {
  const [pending, dispatch] = useReducer(reduce, null);
  const abortRef = useRef<AbortController | null>(null);
  const qc = useQueryClient();

  const send = useCallback(async (query: string) => {
    abortRef.current?.abort();
    const ctrl = new AbortController();
    abortRef.current = ctrl;
    dispatch({ type: 'start' });
    try {
      const stream = await openChatStream({ conversationId: opts.conversationId, query, signal: ctrl.signal });
      for await (const ev of stream) {
        switch (ev.event) {
          case 'trace':
            dispatch({ type: 'trace', trace: ev.data as ReasoningTrace }); break;
          case 'citations':
            dispatch({ type: 'citations', citations: ev.data as Citation[] }); break;
          case 'token':
            dispatch({ type: 'token', delta: (ev.data as { delta: string }).delta }); break;
          case 'done':
            dispatch({ type: 'done' }); break;
          case 'error':
            dispatch({ type: 'error', message: (ev.data as { message: string }).message }); break;
        }
      }
      qc.invalidateQueries({ queryKey: messagesKey(opts.conversationId) });
      qc.invalidateQueries({ queryKey: conversationsKey(opts.topic) });
    } catch (err) {
      if (ctrl.signal.aborted) return;
      const message = err instanceof Error ? err.message : String(err);
      dispatch({ type: 'error', message });
    }
  }, [opts.conversationId, opts.topic, qc]);

  const cancel = useCallback(() => {
    abortRef.current?.abort();
    dispatch({ type: 'reset' });
  }, []);

  return { pending, send, cancel };
}
```

```ts
import type { Message, Citation } from '@buddy/shared';

function citePages(c: Citation): string {
  if (c.pages.length === 0) return c.doc;
  const min = Math.min(...c.pages); const max = Math.max(...c.pages);
  return min === max ? `${c.doc} p.${min}` : `${c.doc} p.${min}-${max}`;
}

export function toMarkdown(messages: Message[], title: string): string {
  const lines: string[] = [`# ${title}`, ''];
  for (const m of messages) {
    lines.push(m.role === 'user' ? '## User' : '## Assistant');
    lines.push('');
    lines.push(m.content);
    if (m.citations.length > 0) {
      lines.push('');
      lines.push('Citations: ' + m.citations.map(citePages).join('; '));
    }
    lines.push('');
  }
  return lines.join('\n');
}

export function toJson(messages: Message[], title: string): string {
  return JSON.stringify({ title, messages }, null, 2);
}

export function downloadBlob(content: string, filename: string, mime: string): void {
  const blob = new Blob([content], { type: mime });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url; a.download = filename;
  document.body.appendChild(a); a.click();
  document.body.removeChild(a);
  URL.revokeObjectURL(url);
}
```

```tsx
export function Composer({ onSubmit, pending }: Props) {
```

```tsx
import { useEffect, useState } from 'react';
import { useNavigate, useParams } from 'react-router-dom';
import { useTopics, useDocs } from '../state/topics.js';
import {
  useConversations, useCreateConversation, useRenameConversation, useDeleteConversation,
} from '../state/conversations.js';
import { useMessages } from '../state/messages.js';
import { useChatStream } from '../state/chat.js';
import { Sidebar } from '../components/Sidebar.js';
import { MessageList } from '../components/MessageList.js';
import { Composer } from '../components/Composer.js';
import { PdfPreview } from '../components/PdfPreview.js';
import { toMarkdown, downloadBlob } from '../lib/export.js';

export default function Chat() {
  const { topic = '' } = useParams();
  const navigate = useNavigate();

  const { data: topics = [] } = useTopics();
  const { data: docs = [] } = useDocs(topic);
  const { data: conversations = [] } = useConversations(topic);
  const create = useCreateConversation(topic);
  const rename = useRenameConversation(topic);
  const del = useDeleteConversation(topic);

  const [activeConvId, setActiveConvId] = useState<string | null>(null);
  useEffect(() => { setActiveConvId(null); }, [topic]);   // reset on topic change

  const { data: messages = [] } = useMessages(activeConvId);
  const { pending, send } = useChatStream({ conversationId: activeConvId ?? '', topic });

  const [preview, setPreview] = useState<{ docId: string; page: number } | null>(null);

  const handleNew = async () => {
    const { id } = await create.mutateAsync(undefined);
    setActiveConvId(id);
  };

  const handleSend = async (q: string) => {
    let cid = activeConvId;
    if (!cid) {
      const { id } = await create.mutateAsync(undefined);
      cid = id; setActiveConvId(id);
    }
    send(q);
  };

  const handleCitationClick = ({ doc, page }: { doc: string; page: number }) => {
    const match = docs.find((d) => d.doc_name === doc);
    if (!match) return;
    setPreview({ docId: match.doc_id, page });
  };

  const handleExport = async (cid: string) => {
    const conv = conversations.find((c) => c.id === cid);
    const msgsForExport = activeConvId === cid ? messages : [];   // simple v1: only active loaded
    downloadBlob(toMarkdown(msgsForExport, conv?.title ?? 'chat'), `${conv?.title ?? 'chat'}.md`, 'text/markdown');
  };

  if (topics.length > 0 && !topics.find((t) => t.topic === topic)) {
    return <p className="p-6">Unknown topic. <button onClick={() => navigate('/')}>Back</button></p>;
  }

  return (
    <div className="flex h-full">
      <Sidebar
        topics={topics}
        activeTopic={topic}
        conversations={conversations}
        activeConvId={activeConvId}
        onNewConversation={handleNew}
        onSelectConversation={setActiveConvId}
        onRenameConversation={(id, title) => rename.mutate({ id, title })}
        onDeleteConversation={(id) => {
          del.mutate(id);
          if (activeConvId === id) setActiveConvId(null);
        }}
        onExportConversation={handleExport}
      />
      <main className="flex flex-1 flex-col">
        <MessageList messages={messages} pending={pending} onCitationClick={handleCitationClick} />
        <Composer onSubmit={handleSend} pending={!!pending && !pending.done} />
      </main>
      {preview && (
        <PdfPreview
          open
          onClose={() => setPreview(null)}
          topic={topic}
          docId={preview.docId}
          initialPage={preview.page}
        />
      )}
    </div>
  );
}
```
